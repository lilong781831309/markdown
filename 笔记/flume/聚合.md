#### 功能

监控文件、监控端口两个agent的数据聚集到第三个agent



#### 配置

taildir-memory-avro.conf

```shell
# Name
a1.sources = r1
a1.sinks = k1
a1.channels = c1

# Source
a1.sources.r1.type = TAILDIR
a1.sources.r1.filegroups = g1 g2
a1.sources.r1.filegroups.g1 = /home/lilong/flume/aggre/taildir/.*\.txt$
a1.sources.r1.filegroups.g2 = /home/lilong/flume/aggre/taildir/.*\.log$
a1.sources.r1.positionFile = /opt/module/flume/conf-file/aggre/taildir_position.json
a1.sources.r1.selector.type = replicating

# Sink
a1.sinks.k1.type = avro
a1.sinks.k1.hostname = hadoop10
a1.sinks.k1.port = 8888

# Channel
a1.channels.c1.type = memory
a1.channels.c1.capacity = 1000
a1.channels.c1.transactionCapacity = 100

# Bind
a1.sources.r1.channels = c1 c2
a1.sinks.k1.channel = c1
a1.sinks.k2.channel = c2
```



netcat-memory-avro.conf

```shell
# Name
a1.sources = r1
a1.sinks = k1
a1.channels = c1

# Source
a1.sources.r1.type = netcat
a1.sources.r1.bind = localhost
a1.sources.r1.port = 6666
a1.sources.r1.selector.type = replicating

# Sink
a1.sinks.k1.type = avro
a1.sinks.k1.hostname = hadoop10
a1.sinks.k1.port = 9999

# Channel
a1.channels.c1.type = memory
a1.channels.c1.capacity = 1000
a1.channels.c1.transactionCapacity = 100

# Bind
a1.sources.r1.channels = c1
a1.sinks.k1.channel = c1
```



avro-memory-logger.conf

```shell
# Name
a1.sources = r1 r2
a1.sinks = k1
a1.channels = c1

# Source
# source1
a1.sources.r1.type = avro
a1.sources.r1.bind = hadoop10
a1.sources.r1.port = 8888
# source2
a1.sources.r2.type = avro
a1.sources.r2.bind = hadoop10
a1.sources.r2.port = 9999

# Sink
a1.sinks.k1.type = logger

# Channel
a1.channels.c1.type = memory
a1.channels.c1.capacity = 1000
a1.channels.c1.transactionCapacity = 100

# Bind
a1.sources.r1.channels = c1
a1.sources.r2.channels = c1
a1.sinks.k1.channel = c1
```



#### 启动

```shell
flume-ng agent --conf $FLUME_HOME/conf --conf-file $FLUME_HOME/conf-file/aggre/avro-memory-logger.conf --name a1 -Dflume.root.logger=INFO,console

flume-ng agent --conf $FLUME_HOME/conf --conf-file $FLUME_HOME/conf-file/aggre/taildir-memory-avro.conf --name a1 -Dflume.root.logger=INFO,console

flume-ng agent --conf $FLUME_HOME/conf --conf-file $FLUME_HOME/conf-file/aggre/netcat-memory-avro.conf --name a1 -Dflume.root.logger=INFO,console
```
